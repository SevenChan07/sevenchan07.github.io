LeetCode-5.最长回文子串

题目描述

给定一个字符串 s，找到 s 中最长的回文子串。你可以假设 s 的最大长度为 1000。

示例1：

输入: "babad"
输出: "bab"
注意: "aba" 也是一个有效答案。

示例2：

输入: "cbbd"
输出: "bb"

解答一

  • 暴力双循环，找出所有子串，选出最长的回文子串。
  • 太慢了。
var longestPalindrome = function (s) {
  if (s.length === 1) return s

  let strTemp = ''
  let maxLength = 1
  let maxSubStr = s.charAt(0)

  for (let i = 0; i < s.length; i++) {
    for (let j = i+1; j < s.length; j++) {
      strTemp = s.slice(i, j+1)
      // 是回文
      if (strTemp === strTemp.split('').reverse().join('')) {
        if (strTemp.length > maxLength) {
          maxLength = strTemp.length
          maxSubStr = strTemp
        }
      }
    }
  }

  return maxSubStr
}

解答二

  • 动态规划
var longestPalindrome = function(s) {
  if (s.length == 1) return s
  
  let arr = []
  for (let i = 0; i < s.length; i ++) {
      arr[i] = []
  };
  
  let begin = 0
  let max = 0

  for(let i = 0; i < s.length; i ++) {
    let j = 0
    while (j <= i) {
      if( s[j] == s[i] && (i - j <= 1 || arr[j+1][i-1] ) ) {
        arr[j][i] = 1
        if (i - j + 1 > max) {
          begin = j
          max = i - j + 1
        }
      }
      j ++
    }
  }
  return s.substr(begin, max)
}

解答三


var longestPalindrome = function(s) {
  if (!s || s.length < 1) {
    return '';
  }

  let strLen = s.length;
  let maxLen = 0; // 最大长度
  let start = 0; // 最大回文串的起始位置

  for (let i = 0; i < s.length; i++) {
    let l = i - 1;
    let r = i + 1;
    let val = s[i];

    while (r <= strLen && s[r] == val) {
      r++;
    }

    while (l >= 0 && r <= strLen && s[l] == s[r]) {
      l--;
      r++;
    }

    let len = r - l - 1;
    if (len > maxLen) {
      maxLen = len;
      start = l;
    }
  }

  return s.substring(start + 1, start + 1+ maxLen);
}

LeetCode-3.无重复字符的最长子串

题目描述

给定一个字符串，请你找出其中不含有重复字符的 最长子串 的长度。

示例1：

输入: "abcabcbb"
输出: 3 
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

示例2：

输入: "bbbbb"
输出: 1
解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

解答一

  • 自己一开始的想法。
  • 两个指针移动，圈定子串。
  • 判断子串是否有重复的。
  • 可以通过。但是效率不高。
var lengthOfLongestSubstring = function (s) {
  if (s.length === 1) return 1
  
  let start = 0
  let end = 1
  let maxLength = 0

  while(start < s.length) {
    while(end < s.length) {
      let temp = s.slice(start, end)
      
      if (temp.indexOf(s.charAt(end)) > -1) {
        // 有重复的
        if (maxLength < (end - start)) {
          maxLength = (end - start)
        }
        break
      } else {
        if (maxLength < (end - start) + 1) {
          maxLength = (end - start) + 1
        }
      }

      end += 1
    }
    start += 1
    end = start + 1
  }

  return maxLength
}

解答二

  • 另一种，更快一点。
var lengthOfLongestSubstring = function(s) {
  let str="" //存放无重复子串
  let size=0 //当前最长无重复子串的长度

  for (let i = 0,len = s.length; i < len; i++) {
    let char = s.charAt(i)
    let index = str.indexOf(char)
    if (index === -1) {
      str += char
      size = size < str.length ? str.length : size
    } else {
        // 去掉重复字符的
      str = str.substr(index + 1) + char
    }
  }
  return size
}

LeetCode-15.三数之和

题目描述

给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有满足条件且不重复的三元组。

注意：答案中不可以包含重复的三元组。

示例：

给定数组 nums = [-1, 0, 1, 2, -1, -4]，

满足要求的三元组集合为：
[
  [-1, 0, 1],
  [-1, -1, 2]
]

解答

  • 不可以用三个循环，那样太慢了。
  • 要考虑不能重复。
  • 先选定第一个数，然后用双指针去遍历剩下两个数，判断三个数加起来是否满足要求。
  • 如果选定的数与相邻的重复，那就跳过。

// 先选定 A ，然后用双指针去遍历剩下的
const threeSum = function (nums) {
  if (nums.length === 0) return []
  
  // 排序
  nums.sort((a, b) => a - b)

  const length = nums.length
  const result = []
  let flag = 0 // 相加为0
  let hash = {}

  // 不可能满足条件
  if (nums[0] > 0 || nums[length - 1] < 0) return []

  for (let i = 0; i < length; i++) {
    // 相同就跳过
    if (nums[i] === nums[i - 1]) continue
    flag = 0 - nums[i]

    // 双指针
    let start = i + 1
    let end = length - 1

    while(start < end) {
      let temp = []
      if (nums[start] + nums[end] > flag) {
        end--
      } else if (nums[start] + nums[end] < flag) {
        start++
      } else {
        temp = [nums[i], nums[start], nums[end]]
        // 判断去重
        if (!hash[temp]) {
          hash[temp] = true
          result.push(temp)
        }

        // 指针移动
        start += 1
        end -=1

        // 去重
        while(start < end && nums[start] === nums[start - 1]) {
          start += 1
        }

        while(start < end && nums[end] === nums[end + 1]) {
          end -= 1
        }
      }
    }

  }

  return result
}
