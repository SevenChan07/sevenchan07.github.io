var longestPalindrome = function(s) {
  if (!s || s.length < 1) {
    return '';
  }

  let strLen = s.length;
  let maxLen = 0; // æœ€å¤§é•¿åº¦
  let start = 0; // æœ€å¤§å›æ–‡ä¸²çš„èµ·å§‹ä½ç½®

  for (let i = 0; i < s.length; i++) {
    let l = i - 1;
    let r = i + 1;
    let val = s[i];

    while (r <= strLen && s[r] == val) {
      r++;
    }

    while (l >= 0 && r <= strLen && s[l] == s[r]) {
      l--;
      r++;
    }

    let len = r - l - 1;
    if (len > maxLen) {
      maxLen = len;
      start = l;
    }
  }

  return s.substring(start + 1, start + 1+ maxLen);
}
